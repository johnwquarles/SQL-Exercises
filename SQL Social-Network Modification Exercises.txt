#Q2  (1/1 point)
#If two students A and B are friends, and A likes B but not vice-versa, remove the Likes tuple. 

delete from Likes
where ID1 in 
    (select Likes.ID1 from Highschooler S1, Highschooler S2, Friend, Likes
     where S1.ID = Friend.ID1 and S2.ID = Friend.ID2 and S1.ID = Likes.ID1 and S2.ID = Likes.ID2
     and S1.ID not in (select ID2 from Likes where ID1 = S2.ID) );
